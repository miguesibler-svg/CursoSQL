create table alumnos(
id_alumno number(4) primary key, 
nombre varchar2(100) not null,
apellido varchar2(100) not null,
telefono number(10) not null,
fecha_ingreso date not null
);

create table materias(
id_materia number(4) primary key,
nombre_materia varchar2(100) not null,
nota1 number(3,1) not null,
nota2 number(3,1) not null,
nota3 number(3,1) not null, 
calificacion_final number(3,1),
profesor varchar2(10) not null,
id_alumno number(4) not null, 
constraint fk_id_alumno
foreign key(id_alumno)
references alumnos(id_alumno)
);
