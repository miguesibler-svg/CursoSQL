ejercicios de creación, alteración y eliminación de tablas

/* Ejercicio 1
Crea la tabla clientes con nombre VARCHAR2(80) y ciudad VARCHAR2(50).
Luego agrega la columna telefono VARCHAR2(20).
Finalmente elimina la tabla clientes.
*/

create table clientes(
    nombre varchar2(80),
    ciudad varchar2(50)
);

alter table clientes 
add telefono varchar(20);

drop table clientes;

/* Ejercicio 2
Crea la tabla autos con marca VARCHAR2(50) y modelo VARCHAR2(50).
Luego agrega la columna precio NUMBER(10,2).
Después modifica modelo para que permita 80 caracteres.
Finalmente elimina la tabla autos.
*/

create table autos(
    marca varchar2(50),
    modelo varchar2(50)
);

alter table autos
modify modelo varchar2(80);

drop table autos;
