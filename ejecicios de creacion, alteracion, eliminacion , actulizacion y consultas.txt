-- Obtener el precio máximo y mínimo de pedidos
-- donde el id_mesero esté en (1, 3, 5)
-- y el precio sea mayor a 100.
select max(precio), min(precio)
from pedidoss
where id_mesero in (1,3,4) and precio>100;
-- Mostrar pedidos de diferentes meseros
-- donde el id_mesero esté en (1, 2, 4)
-- y la cantidad sea mayor a 1.
select *
from pedidoss
where id_mesero in(1,2,4) and total_platillos>1;
-- Mostrar los pedidos
-- donde la cantidad esté en (1, 2, 4)
-- y el precio sea mayor a 100.
select  *
from pedidoss
where total_platillos in (1,2,4) and precio>100;
-- Ordenar pedidos por precio de mayor a menor
-- donde el id_mesero esté en (2, 4)
-- y la cantidad sea mayor a 1.
select *
from pedidoss
where id_mesero in (2,4) and total_platillos>1
order by precio desc;
-- Agrupar pedidos por id_mesero
-- donde el precio esté en (90, 120, 150)
-- y la cantidad sea mayor a 1.
select id_mesero, count(*)
from pedidoss
where precio in (90,120,150) and total_platillos>1
group by id_mesero;
