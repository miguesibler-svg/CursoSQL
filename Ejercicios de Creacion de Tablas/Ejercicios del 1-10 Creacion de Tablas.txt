-- EJERCICIO 1

-- REALIZAR:
-- CREATE TABLE CLIENTES CON LLAVE PRIMARIA.
-- CREATE TABLE PEDIDOS CON LLAVE PRIMARIA Y LLAVE FORÁNEA HACIA CLIENTES.

-- LA BASE DE DATOS DEBE CONTENER LAS SIGUIENTES TABLAS:

-- 1. TABLA CLIENTES
-- id_cliente
-- nombre
-- correo
-- telefono
create table clientes(
id_cliente number(10) primary key,
nombre varchar2(100) not null,
correo varchar2(100) not null,
telefono number(10) not null
);
-- 2. TABLA PEDIDOS
-- id_pedido
-- fecha_pedido
-- total
-- id_cliente
create table pedidos(
id_pedido number(10) generated always as identity primary key,
fecha_pedido date not null,
total number(10) not null,
id_cliente number(10) not null,
constraint fk_id_cliente
foreign key(id_cliente)
references clientes(id_cliente)
);
