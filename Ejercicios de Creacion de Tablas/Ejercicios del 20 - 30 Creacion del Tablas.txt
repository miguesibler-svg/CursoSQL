-- EJERCICIO 21

-- REALIZAR:
-- CREATE TABLE ESCUELAS CON LLAVE PRIMARIA.
-- CREATE TABLE MAESTROS CON LLAVE PRIMARIA Y LLAVE FORÁNEA HACIA ESCUELAS.

-- LA BASE DE DATOS DEBE CONTENER LAS SIGUIENTES TABLAS:

-- 1. TABLA ESCUELAS
-- id_escuela
-- nombre_escuela
-- direccion
create table escuelas(
id_escuela number(10) primary key,
nombre_escuela varchar2(100) not null,
direccion varchar2(100) not null
);
-- 2. TABLA MAESTROS
-- id_maestro
-- nombre
-- especialidad
-- antiguedad
-- id_escuela
create table maestros(
id_maestro number(10) primary key,
nombre varchar2(100) not null,
especialidad varchar2(100) not null,
antiguedad varchar2(100) not null,
id_escuela number(10) not null,
constraint fk_id_escuela
foreign key(id_escuela)
references escuelas(id_escuela)
);
