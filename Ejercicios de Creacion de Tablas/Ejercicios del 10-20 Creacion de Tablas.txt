-- REALIZAR:
-- CREATE TABLE CARRERAS CON LLAVE PRIMARIA.
-- CREATE TABLE ALUMNOS CON LLAVE PRIMARIA Y LLAVE FORÁNEA HACIA CARRERAS.

-- LA BASE DE DATOS DEBE CONTENER LAS SIGUIENTES TABLAS:

-- 1. TABLA CARRERAS
-- id_carrera
-- nombre_carrera
-- facultad
create table carreras(
id_carrera number(10) primary key,
nombre_carrera varchar2(10) not null,
facultad varchar2(100) not null
);
-- 2. TABLA ALUMNOS
-- id_alumno
-- nombre
-- edad
-- semestre
-- id_carrera
create table alumnos(
id_alumno number(10) primary key,
nombre varchar2(100) not null,
edad number(10) not null,
semestre varchar2(100) not null,
id_carrera number(10) not null,
constraint fk_id_carrera
foreign key(id_carrera)
references carreras(id_carrera)
);
