ejercicios de primaria y foráneas 
/* Ejercicio 1
Crea la tabla clientes con:
id_cliente NUMBER como clave primaria,
nombre VARCHAR2(80).

Crea la tabla pedidos con:
id_pedido NUMBER como clave primaria,
id_cliente NUMBER como clave foránea que referencia a clientes(id_cliente),
fecha DATE.
*/

create table clientes(
    id_cliente number(10) primary key,
    nombre varchar2(80)
);

create table pedidos(
    id_pedido number(10) primary key,
    id_cliente number(10),
    
    constraint fk_id_cliente
    foreign key (id_cliente)
    references clientes(id_cliente)
);

/* Ejercicio 3

alter table departamentos

create table empleados(
id_empleado number(5),

nombre varchar2(80),

constraint fk_id_departamento
