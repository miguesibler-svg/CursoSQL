-- LA BASE DE DATOS DEBE CONTENER LAS SIGUIENTES TABLAS:

-- 1. TABLA CLIENTES
-- id_cliente
-- nombre
-- ciudad
create table clientes(
id_cliente number(10) primary key,
nombre varchar2(100),
ciudad varchar2(100)
);

-- 2. TABLA PEDIDOS
-- id_pedido
-- descripcion
-- cantidad
-- categoria
-- total
-- id_cliente
create table pedidos(
id_pedido number(10) primary key,
descripcion varchar2(100),
cantidad number(10),
categoria varchar2(100),
total number(10),
id_cliente number(10),
constraint fk_id_cliente
foreign key(id_cliente)
references clientes(id_cliente)
);

-- CONSULTA 1
-- Mostrar el ID del pedido, descripcion, cantidad, categoria y total.
-- Ademas, realizar una concatenacion entre la descripcion del pedido y la categoria,
-- mostrar la longitud de la descripcion,
-- calcular la suma total de los montos,
-- mostrar el total maximo, minimo y promedio general,
-- asi como la cantidad total de pedidos encontrados.
-- Considerar solamente los pedidos cuyas descripciones contengan la letra 'o',
-- que pertenezcan a las categorias de Tecnologia, Muebles o Hogar,
-- que tengan un total entre 4000 y 30000,
-- excluyendo la categoria Electrodomesticos.
-- Ordenar los resultados de mayor a menor total.
select id_pedido, descripcion, cantidad, total, descripcion||'-'||categoria, 
       length(descripcion),sum(total), max(total), min(total), avg(total),count(*)
from pedidos
where descripcion like '%o%' and categoria in ('Tecnologia', 'Muebles' ,'Hogar') and total between 4000 and 30000 
and not categoria='Electrodomesticos'
group by id_pedido, descripcion, cantidad, total, categoria
order by total desc;
