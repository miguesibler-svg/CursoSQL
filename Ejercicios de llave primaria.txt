Ejercicios de llave primaria

/* Ejercicio 1
Crea la tabla alumnos con:
id_alumno NUMBER como clave primaria,
nombre VARCHAR2(80),
edad NUMBER.
*/
create table alumnos(
id_alumno number(10) primary key,
nombre varchar2(80),
edad number(2)
);

/* Ejercicio 2
Crea la tabla productos con:
id_producto NUMBER como clave primaria,
nombre VARCHAR2(80),
precio NUMBER(10,2).
*/
create table productos(
    id_productos number(10,0) primary key,
    nombre varchar2(80),
    precio number(10,2)
);

/* Ejercicio 3
Crea la tabla empleados con:
id_empleado NUMBER como clave primaria,
nombre VARCHAR2(80),
salario NUMBER(10,2).
*/
create table empleados(
    id_empleado number(10) primary key,
    nombre varchar2(50),
    salario number(10,2)
);

/* Ejercicio 4

/* Ejercicio 6
Crea la tabla clientes con:
id_cliente NUMBER,
nombre VARCHAR2(80).
Luego agrega la clave primaria a id_cliente con ALTER TABLE.
*/

create table clientes(
    id_cliente number(10),
    nombre varchar2(80)
);

alter table clientes
add primary key (id_cliente);
