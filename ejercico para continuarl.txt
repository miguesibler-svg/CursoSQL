-- La base de datos debe contener las siguientes tablas:

-- 1. TABLA DOCTORES
-- id_doctor
-- nombre
-- especialidad
create table doctores(
id_doctor number(10) primary key,
nombre varchar2(100) not null,
especialidad varchar2(100) not null
);

-- 2. TABLA PACIENTES
-- id_paciente
-- nombre
-- edad
-- diagnostico
-- id_doctor
create table pacientes(
id_paciente number(10) primary key,
nombre varchar2(100) not null,
edad number(10) not null,
diagnostico varchar2(100) not null,
id_doctor number(10) not null,
constraint fk_id_pacientes
foreign key(id_doctor)
references doctores(id_doctor)
);

-- CONSULTAS

-- 1. Mostrar los nombres concatenados con la especialidad de 
--los doctores donde la especialidad sea 'Cardiologia' o 'Neurologia' 
--y el nombre empiece con 'A', agrupando por especialidad, mostrando el MAX y MIN del nombre, 
--la cantidad COUNT de doctores, ordenando por nombre ascendente.

select nombre||' '||especialidad, max(nombre), min(nombre), count(*)
from doctores 
where (especialidad='Cardiologia' or especialidad='Neurologia') and nombre like 'A%'
group by especialidad,nombre
order by nombre;

-- 2. Mostrar los pacientes cuyo diagnostico sea 'Gripe' o 'Migraña' 
--y cuya edad sea mayor a 20 AND menor a 40, mostrando la edad + 5, 
--agrupando por diagnostico, mostrando MAX y MIN de edad, COUNT 
--de pacientes y ORDER BY nombre ASC.
select enfermedad, max(edad), min(edad), count(*), max(edad+5)
from enfermos
where (enfermedad='Gripe' or enfermedad='Migraña') and edad >20 and edad<40
group by enfermedad, edad, nombre
order by nombre;

-- 3. Mostrar los doctores cuyo id_doctor IN (1,2,3) y cuyo nombre LIKE '%ez%',
--agrupando por especialidad, mostrando MAX y MIN de nombre, COUNT de doctores 
--y ORDER BY nombre ASC.

select nombre, max(nombre), min(nombre), count(*)
from doctores
where id_doctor in (1,2,3) and nombre like '%ez%'
group by especialidad, nombre
order by nombre;

-- 4. Mostrar los pacientes cuyo nombre LIKE '%ar%' y cuya edad 
--sea mayor a 15 AND menor a 50, mostrando edad * 2, agrupando
--por id_doctor, mostrando MAX y MIN de edad, COUNT y ORDER BY edad DESC.

select nombre, edad*2, max(edad), min(edad), count(*)
from enfermos
where nombre like '%r%' and edad >15 and edad<40
group by nombre, edad
order by edad desc;

-- 5. Mostrar los doctores cuya especialidad sea 'Pediatria' OR 'Dermatologia',
--y cuyo nombre LIKE 'M%', agrupando por especialidad, mostrando COUNT, MAX y MIN 
--de nombre, ORDER BY especialidad ASC.
select nombre, count(*), max(nombre), min(nombre)
from doctores
where (especialidad='Pediatria' or especialidad='Dermatologia') and nombre like 'M%'
group by especialidad,nombre
order by especialidad;

-- 6. Mostrar los pacientes cuyo id_paciente IN (101,102,105) y 
--cuyo diagnostico LIKE '%a%', mostrando edad + 10, agrupando por diagnostico, 
--mostrando MAX y MIN de edad, COUNT y ORDER BY nombre ASC.

select nombre, enfermedad, edad+10, max(edad), min(edad), count(*)
from enfermos 
where id_paciente in ('101','102','105') and enfermedad like '%a%'
group by enfermedad,nombre,edad
order by nombre;

-- 7. Mostrar los doctores cuyo nombre LIKE '%o%' y cuya especialidad 
--sea 'Cardiologia' OR 'Cirugia', agrupando por especialidad, mostrando
--MAX y MIN de nombre, COUNT y ORDER BY nombre DESC.
select nombre, especialidad, length(nombre), min(nombre), count(*)
from doctores
where nombre like '%o%' and (especialidad='Cardiologia' or especialidad='cirugia')
group by especialidad,nombre
order by nombre desc;

-- 8. Mostrar los pacientes cuyo nombre empiece con 'D' y cuya edad sea mayor 
--a 18 AND menor a 60, mostrando edad - 2, agrupando por id_doctor, mostrando 
--MAX y MIN de edad, COUNT y ORDER BY edad ASC.

select id_doctor,nombre, edad-2, max(edad), min(edad)
from enfermos
where nombre like '%D%' and edad > 18 and edad <60
group by id_doctor,nombre,edad
order by edad;

-- 9. Mostrar los doctores cuyo id_doctor IN (2,4,5) y cuyo nombre LIKE '%a%',
--agrupando por especialidad, mostrando COUNT, MAX y MIN de nombre, ORDER BY nombre ASC.

select nombre, especialidad, count(*), max(nombre), min(nombre)
from doctores
where id_doctor in (2,4,5) and nombre like '%a%'
group by nombre,especialidad
order by nombre;

-- 10. Mostrar los pacientes cuyo diagnostico sea 'Dolor' OR 'Infeccion' 
--y cuya edad sea mayor a 25 AND menor a 50, mostrando edad * 3, agrupando 
--por diagnostico, mostrando MAX y MIN de edad, COUNT y ORDER BY nombre DESC.

select nombre, edad*3, count(*), length(max(nombre)),length(min(nombre)) 
from enfermos
where (enfermedad='Dolor' or enfermedad='Infeccion') and edad >25 and edad<50
group by enfermedad, nombre, edad
order by nombre desc;
