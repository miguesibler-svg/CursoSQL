/* 1. Mostrar el nombre concatenado con la ciudad de los clientes que sean de CDMX 
o Guadalajara, cuyo nombre empiece con 'J', agrupando por ciudad, mostrando la 
cantidad de clientes, el nombre máximo y mínimo en orden ascendente por ciudad. */

select concat(nombre,ciudad), count(*), max(nombre), min(nombre)
from clientes 
where ciudad in ('CDMX','Guadalajara') and  nombre like 'J%'
group by ciudad, nombre
order by ciudad;

/* 2. Mostrar los productos concatenando el nombre del producto y el precio, donde 
el stock sea mayor a 10 y menor a 50, o el nombre del producto contenga 'Samsung', 
agrupando por stock, mostrando el precio máximo, mínimo y el conteo de productos, 
ordenando de mayor a menor precio. */

select nombre_producto||' '||precio, max(precio), min(precio), count(*)
from productos
where stock >10 and stock < 50 and nombre_producto like '%Samsung%'
group by nombre_producto, precio
order by precio desc;

/* 3. Mostrar los correos concatenados con el teléfono de los clientes que vivan en 
Puebla o Monterrey y cuyo correo termine en 'email.com', agrupando por ciudad, 
mostrando la cantidad de clientes, el teléfono máximo y mínimo, ordenando por nombre. */

select correo||' '||telefono, count(*), max(telefono), min(telefono)
from clientes
where ciudad in ('Puebla','Monterrey') and correo like '%email.com%' 
group by correo, telefono
order by correo;

/* 4. Mostrar los nombres de productos concatenados con el stock donde el precio sea 
mayor a 1000 y el nombre contenga la letra 'o', o el stock sea menor a 20, agrupando 
por precio, mostrando el máximo y mínimo stock y el conteo de productos, ordenando 
por stock descendente. */

select nombre_producto||' '||stock, max(stock), min(stock) 
from productos
where precio >1000 and nombre_producto like '%o%' or stock<20
group by nombre_producto, stock
order by stock desc;

/* 5. Mostrar los nombres concatenados con los correos de los clientes donde el id_cliente 
sea mayor a 5 y menor a 15, o la ciudad contenga la letra 'a', agrupando por ciudad, 
mostrando el id máximo, mínimo y el total de clientes, ordenando por id_cliente. */

select nombre||' '||correo, max(id_cliente), min(id_cliente)
from clientes
where id_cliente > 5 and id_cliente<15 or ciudad like '%a%'
group by nombre,correo
order by nombre;

/* 6. Mostrar los productos concatenados con el precio donde el nombre del producto 
empiece con 'M' y el stock sea mayor a 5, o el precio sea menor a 2000, agrupando 
por stock, mostrando el precio máximo, mínimo y la cantidad de productos, ordenando 
por precio ascendente. */

select nombre_producto||' '||precio, max(precio), min(precio)
from productos
where nombre_producto like 'M%' and stock >5 or precio >2000
group by nombre_producto, precio
order by nombre_producto desc;

/* 7. Mostrar los clientes concatenando nombre y teléfono donde la ciudad sea León 
o Toluca y el nombre contenga la letra 'e', agrupando por ciudad, mostrando el id 
máximo, mínimo y el conteo de clientes, ordenando por nombre descendente. */

select nombre||' '||telefono, max(id_cliente), min(id_cliente), count(*)
from clientes
where ciudad in ('Leon','Toluca') and nombre like '%e%'
group by nombre, telefono, ciudad
order by nombre;

/* 8. Mostrar los nombres de productos concatenados con el stock donde el precio sea 
mayor a 500 y menor a 15000, y el nombre contenga la letra 'a', o el stock sea mayor 
a 30, agrupando por precio, mostrando el stock máximo, mínimo y el conteo de productos, 
ordenando por nombre_producto. */

select nombre_producto||' '||stock, max(stock), min(stock), count(*),precio
from productos
where precio > 500 and precio<15000 and nombre_producto like '%a%' or stock > 30
group by nombre_producto, stock, precio
order by precio desc;

/* 9. Mostrar los nombres concatenados con las ciudades de los clientes donde el correo 
contenga 'email' y el teléfono termine en '1', o el id_cliente sea menor a 10, agrupando 
por ciudad, mostrando el id máximo, mínimo y la cantidad de clientes, ordenando por ciudad. */

select nombre||' '||ciudad, max(id_cliente), min(id_cliente), count(*)
from clientes
where correo like '%email%' and telefono like '%1' or id_cliente < 10 
group by ciudad, nombre
order by nombre;

/* 10. Mostrar los productos concatenando nombre_producto y precio donde el stock sea 
mayor a 5 y el nombre contenga la letra 'r', o el precio sea mayor a 10000, agrupando 
por stock, mostrando el precio máximo, mínimo y el conteo de productos, ordenando 
por precio descendente. */

select stock, nombre_producto||' '||precio, max(precio), min(precio), count(*)
from productos
where stock>5 and nombre_producto like '%r%' or precio>10000
group by nombre_producto, precio, stock
order by stock desc;
